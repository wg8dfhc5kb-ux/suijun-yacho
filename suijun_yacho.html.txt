<script>
let rows = [
  {no:'1', point:'BM.1', bs:'', fs:'1.235', memo:''},
  {no:'2', point:'TP.1', bs:'', fs:'1.410', memo:''},
  {no:'3', point:'TP.2', bs:'', fs:'0.985', memo:''}
];

document.getElementById('date').value = new Date().toISOString().slice(0,10);

function n(v){ const x = Number(String(v ?? '').replace(/,/g,'')); return Number.isFinite(x) ? x : 0; }
function f3(v){ return n(v).toLocaleString('ja-JP',{minimumFractionDigits:3, maximumFractionDigits:3}); }
function saveLocal(){
  const data = {project:project.value, date:date.value, observer:observer.value, rows};
  localStorage.setItem('suijun_yacho_data', JSON.stringify(data));
}
function loadLocal(){
  try{
    const data = JSON.parse(localStorage.getItem('suijun_yacho_data') || 'null');
    if(data){ project.value=data.project||'水準測量 野帳'; date.value=data.date||date.value; observer.value=data.observer||''; rows=Array.isArray(data.rows)?data.rows:rows; }
  }catch(e){}
}
function render(){
  const tbody = document.getElementById('tbody'); tbody.innerHTML = '';
  rows.forEach((r,i)=>{
    const tr=document.createElement('tr');
    tr.innerHTML = `
      <td class="no"><input value="${esc(r.no)}" oninput="upd(${i},'no',this.value)"></td>
      <td class="point"><input value="${esc(r.point)}" placeholder="BM / TP等" oninput="upd(${i},'point',this.value)"></td>
      <td class="num"><input inputmode="decimal" value="${esc(r.bs)}" placeholder="0.000" oninput="upd(${i},'bs',this.value)"></td>
      <td class="num fs"><input inputmode="decimal" value="${esc(r.fs)}" placeholder="0.000" oninput="upd(${i},'fs',this.value)"></td>
      <td><input value="${esc(r.memo)}" placeholder="メモ" oninput="upd(${i},'memo',this.value)"></td>
      <td class="del"><button class="danger" onclick="delRow(${i})" ${rows.length===1?'disabled':''}>削除</button></td>`;
    tbody.appendChild(tr);
  });
  calc(); saveLocal();
}
function esc(s){ return String(s ?? '').replaceAll('&','&amp;').replaceAll('<','&lt;').replaceAll('>','&gt;').replaceAll('"','&quot;'); }
function upd(i,k,v){ rows[i][k]=v; calc(); saveLocal(); }
function calc(){
  const fs = rows.reduce((a,r)=>a+n(r.fs),0), bs = rows.reduce((a,r)=>a+n(r.bs),0);
  const cnt = rows.filter(r=>String(r.fs||'').trim() !== '').length;
  fsTotal.textContent=f3(fs); fsFoot.textContent=f3(fs); bsTotal.textContent=f3(bs); bsFoot.textContent=f3(bs); fsCount.textContent=cnt; diffTotal.textContent=f3(bs-fs);
}
function addPoint(){ rows.push({no:String(rows.length+1), point:'TP.'+rows.length, bs:'', fs:'', memo:''}); render(); setTimeout(()=>document.querySelector('#tbody tr:last-child .point input')?.focus(),50); }
function delRow(i){ rows.splice(i,1); rows.forEach((r,idx)=>r.no=String(idx+1)); render(); }
function clearBook(){ if(confirm('入力内容をクリアしますか？')){ rows=[{no:'1',point:'',bs:'',fs:'',memo:''}]; render(); } }
function csvCell(v){ return '"'+String(v ?? '').replaceAll('"','""')+'"'; }
function exportCsv(){
  const lines=[];
  lines.push(['件名',project.value].map(csvCell).join(',')); lines.push(['測量日',date.value].map(csvCell).join(',')); lines.push(['観測者',observer.value].map(csvCell).join(',')); lines.push('');
  lines.push(['No','測点','BS','FS','摘要'].map(csvCell).join(','));
  rows.forEach(r=>lines.push([r.no,r.point,r.bs,r.fs,r.memo].map(csvCell).join(',')));
  lines.push(''); lines.push(['','合計',bsTotal.textContent,fsTotal.textContent,'BS-FS '+diffTotal.textContent].map(csvCell).join(','));
  const blob=new Blob(['\ufeff'+lines.join('\n')],{type:'text/csv;charset=utf-8;'});
  const a=document.createElement('a'); a.href=URL.createObjectURL(blob); a.download=(project.value||'水準測量野帳')+'_'+date.value+'.csv'; a.click(); URL.revokeObjectURL(a.href);
}
['project','date','observer'].forEach(id=>document.getElementById(id).addEventListener('input',saveLocal));
loadLocal(); render();
</script>
